# Описание структуры данных атрнибута

(Пример основан на атрибуте target)

```json
{
    // Имя атрибута
    "name": "target",
    // Основаная информация хранится в свойстве structure. Это массив из объектов со свойствами type и value
    //
    //  structure: [
    //      {
    //          "type": string;
    //          "value": any;
    //      }
    //  ]
    //
    "structure": [
        // Категория/тип атрибута
        {
            "type": "type",
            "value": "global"
        },
        //
        // description - Краткое описание назначения атрибута.
        // value - строка с jsx элементами (прогоняется через интерпритатор при рендеренге)
        //
        {
            "type": "description",
            "value": "В стандартном режиме, при клике по ссылке, новая станица открывается в том же окне, заменяя предыдущую страницу. Разные значения атрибута <LA>target</LA> могут изменить данное поведение. Например, при клике на ссылку, новая страница может открываться в соседнем окне."
        },
        //
        // values - блок описывает возможное значения атрибута - данные будут отображены в таблице
        // description - краткое описание - строка
        // values - массив объектов с полями name, meaning где name имя значения и meaning - краткое описание значения
        // defaultValue - значение атрибута по умолчанию
        //
        {
            "type": "values",
            "value": {
                "description": "Используйте данные значения атрибута чтобы переопределить поведение при переходе по ссылке.",
                "values":  [
                    {
                        "name": "_self",
                        "meaning": "Загружает страницу в текущую вкладку/окно браузера."
                    },
                    {
                        "name": "_blank",
                        "meaning": "Загружает страницу в новую вкладку/окно браузера."
                    },
                    {
                        "name": "_parent",
                        "meaning": "Открывает страницу в родительском фрейме."
                    },
                    {
                        "name": "_top",
                        "meaning": "Открывает страницу во всем теле окна."
                    }
                ],
                "defaultValue": "_self"
            }
        },
        {
            "type": "fullDescription",
            "value": true
        }
    ]
}
```
